

import cv2
import numpy as np
import matplotlib.pyplot as plt

# ğŸ“· Load the image in grayscale
image = cv2.imread(r'C:\Users\SHREEKSHITHA\Pictures\nature.jpg', cv2.IMREAD_GRAYSCALE)
if image is None:
    raise FileNotFoundError("The image file was not found.")

# ğŸŒ«ï¸ Apply Gaussian Blur (smooths image, reduces noise)
gaussian_blur = cv2.GaussianBlur(image, (5, 5), 0)

# ğŸ§½ Apply Median Blur (good for salt-and-pepper noise)
median_blur = cv2.medianBlur(image, 5)

# ğŸ§  Apply Laplacian Filter (for edge detection)
laplacian = cv2.Laplacian(image, cv2.CV_64F)
laplacian = cv2.convertScaleAbs(laplacian)  # Convert to 8-bit image

# ğŸ“Š Display all filtered images
titles = ['Original Image', 'Gaussian Blur', 'Median Blur', 'Laplacian Filter']
images = [image, gaussian_blur, median_blur, laplacian]

plt.figure(figsize=(14, 6))
for i in range(4):
    plt.subplot(1, 4, i + 1)
    plt.imshow(images[i], cmap='gray')
    plt.title(titles[i])
    plt.axis('off')

plt.tight_layout()
plt.show()
